```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>SOAP病历智能评分系统</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            margin: 20px;
        }
        table {
            width: 100%;
            border-collapse: collapse;
            margin-top: 20px;
        }
        th, td {
            border: 1px solid #ddd;
            padding: 8px;
            text-align: left;
        }
        th {
            background-color: #f2f2f2;
        }
    </style>
</head>
<body>

<h1>SOAP病历智能评分系统</h1>

<div id="subjective"></div>
<div id="objective"></div>
<div id="assessment"></div>
<div id="plan"></div>

<script>
    // 主观评分数据
    const subjectiveData = [
        { item: '格式规范性', score: 5 },
        { item: '主诉完整性', score: 2 },
        { item: '症状演变', score: 5 },
        { item: '诊治经过', score: 3 },
        { item: '相关病史', score: 3 },
        { item: '家族史', score: 2 },
        { item: '社会心理', score: 5 }
    ];

    // 客观评分数据
    const objectiveData = [
        { item: '体格检查', score: 6 },
        { item: '实验室检查', score: 4 },
        { item: '其他评估', score: 0 }
    ];

    // 评估评分数据
    const assessmentData = [
        { item: '主要诊断', score: 7 },
        { item: '危险因素', score: 9 },
        { item: '并发症', score: 4 },
        { item: '依从性', score: 2 },
        { item: '家庭资源', score: 1 }
    ];

    // 处置计划评分数据
    const planData = [
        { item: '检查计划', score: 8 },
        { item: '药物治疗', score: 9 },
        { item: '非药物治疗', score: 14 },
        { item: '随诊要求', score: 3 }
    ];

    // 计算总分
    function calculateTotalScore(data) {
        return data.reduce((sum, item) => sum + item.score, 0);
    }

    // 渲染主观评分
    function renderSubjective() {
        const subjectiveDiv = document.getElementById('subjective');
        subjectiveDiv.innerHTML += `<ul>`;
        subjectiveData.forEach(item => {
            subjectiveDiv.innerHTML += `<li>${item.item}: ${item.score}/5 → ${item.description}</li>`;
        });
        subjectiveDiv.innerHTML += `</ul>`;
        subjectiveDiv.innerHTML += `<p>主观评分总分 = ${calculateTotalScore(subjectiveData)}/25分</p>`;
    }

    // 渲染客观评分
    function renderObjective() {
        const objectiveDiv = document.getElementById('objective');
        objectiveDiv.innerHTML += `<ul>`;
        objectiveData.forEach(item => {
            objectiveDiv.innerHTML += `<li>${item.item}: ${item.score}/8 → ${item.description}</li>`;
        });
        objectiveDiv.innerHTML += `</ul>`;
        objectiveDiv.innerHTML += `<p>客观评分总分 = ${calculateTotalScore(objectiveData)}/15分</p>`;
    }

    // 渲染评估评分
    function renderAssessment() {
        const assessmentDiv = document.getElementById('assessment');
        assessmentDiv.innerHTML += `<ul>`;
        assessmentData.forEach(item => {
            assessmentDiv.innerHTML += `<li>${item.item}: ${item.score}/8 → ${item.description}</li>`;
        });
        assessmentDiv.innerHTML += `</ul>`;
        assessmentDiv.innerHTML += `<p>评价评分总分 = ${calculateTotalScore(assessmentData)}/25分</p>`;
    }

    // 渲染处置计划评分
    function renderPlan() {
        const planDiv = document.getElementById('plan');
        planDiv.innerHTML += `<ul>`;
        planData.forEach(item => {
            planDiv.innerHTML += `<li>${item.item}: ${item.score}/6 → ${item.description}</li>`;
        });
        planDiv.innerHTML += `</ul>`;
        planDiv.innerHTML += `<p>处置计划总分 = ${calculateTotalScore(planData)}/35分</p>`;
    }

    // 渲染所有评分
    renderSubjective();
    renderObjective();
    renderAssessment();
    renderPlan();
</script>

</body>
</html>
```

这个HTML页面展示了SOAP病历智能评分系统的各个分项评分，并将它们合并计算出总分。每个分项都以`<li>`形式列出，方便阅读。
